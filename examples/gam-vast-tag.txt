# The Nexus Engine - Catalyst GAM VAST Tag (Publisher Enrichments)
#
# This template lives in the repo. {GVL_ID} is our own registered IAB TCF
# vendor ID - a static constant, baked into the tag once at template level.
# Publishers never see it; when a CMP on the publisher's page has us in its
# vendor list, GAM emits the full signed TCF consent string via
# %%GDPR_CONSENT_{GVL_ID}%%. Catalyst decodes that string server-side and
# applies per-bidder TCF GVL checks before fanning out to SSPs - no further
# publisher action required.
#
# Before emailing a copy of this URL to a publisher:
#   {GVL_ID}   -> replace with our numeric GVL ID (one-time, repo-wide).
#
# Publisher fills in when trafficking:
#   {PUBLISHER_ID}  -> Nexus Engine publisher ID (e.g. pub-video-123456)
#   {MIN} / {MAX}   -> allowed creative duration in seconds (e.g. 5 / 30)
#   {FLOOR}         -> USD bid floor (omit bidfloor= to disable)
#
# Endpoint: ads.thenexusengine.com is the live Catalyst host (no separate
# test subdomain - staging runs behind the same hostname via nginx traffic
# split). Use a test publisher ID during the initial 1,000-request window
# and switch to production once validation passes.
#
# ---------------------------------------------------------------------------

https://ads.thenexusengine.com/video/vast?pub_id={PUBLISHER_ID}&w=%%WIDTH%%&h=%%HEIGHT%%&mindur={MIN}&maxdur={MAX}&mimes=video/mp4,application/x-mpegURL&protocols=2,3,5,6,7,8&placement=1&linearity=1&bidfloor={FLOOR}&placement_id=%%ADUNIT%%&page_url=%%PAGE_URL%%&domain=%%SITE_DOMAIN%%&ref=%%REFERRER_URL%%&cb=%%CACHEBUSTER%%&sport=%%PATTERN:sport%%&competition=%%PATTERN:competition%%&lang=%%PATTERN:language%%&device_type=%%PATTERN:device%%&geo=%%PATTERN:geo%%&content_type=%%PATTERN:content_type%%&gdpr=%%GDPR%%&gdpr_consent=%%GDPR_CONSENT_{GVL_ID}%%&addtl_consent=%%ADDTL_CONSENT%%&us_privacy=%%US_PRIVACY%%&gpp=%%GPP_STRING%%&gpp_sid=%%GPP_SID%%&coppa=%%TFCD%%&ifa=%%ADVERTISING_IDENTIFIER_PLAIN%%&ifa_type=%%ADVERTISING_IDENTIFIER_TYPE%%&lmt=%%LIMITADTRACKING%%

# ---------------------------------------------------------------------------
# OpenRTB 2.5 mapping (reference)
#
# VAST Enrichments (standard macros)
#   page_url        -> site.page
#   domain          -> site.domain
#   ref             -> site.ref
#   cb              -> (not carried; forces fresh fetch)
#   placement_id    -> imp[0].tagid                          (Ad unit path)
#   w / h           -> imp[0].video.w / imp[0].video.h
#
# Custom Key-Values (contextual targeting)
#   sport           -> site.content.genre  + imp.ext.context.sport
#   competition     -> site.content.series + imp.ext.context.competition
#   lang            -> site.content.language (BCP-47)
#   device_type     -> device.devicetype (OpenRTB enum)
#   geo             -> device.geo.country (ISO 3166-1 alpha-3)
#   content_type    -> site.content.cattax + site.content.cat[]
#
# Privacy Signals (GAM native, passed through where applicable)
#   gdpr            -> regs.ext.gdpr
#   gdpr_consent    -> user.ext.consent
#   addtl_consent   -> user.ext.ConsentedProvidersSettings.consented_providers
#   us_privacy      -> regs.ext.us_privacy
#   gpp / gpp_sid   -> regs.gpp / regs.gpp_sid[]
#   coppa           -> regs.coppa
#
# User Signals (privacy-compliant identifiers via GAM)
#   ifa / ifa_type  -> device.ifa / device.ext.ifa_type
#   lmt             -> device.lmt
#
# NOT in this tag (per publisher scope):
#   - Deterministic first-party IDs (UID2 / ID5 / LiveRamp / SharedID / PPID).
#     When the publisher's 1p strategy is ready, switch that line item to
#     POST /video/openrtb and populate user.ext.eids[] - the GET tag cannot
#     carry structured EID arrays.
#   - supply chain object (source.ext.schain) - handled server-side by
#     Catalyst, not injected from the publisher side.
